end

local function colorClose(c1, c2, tol)
    tol = tol or 0.01
    return math.abs(c1.R-c2.R)<tol and math.abs(c1.G-c2.G)<tol and math.abs(c1.B-c2.B)<tol
end

            if ok then
                task.wait(1.5)
                local snpcs = workspace:FindFirstChild("ServerNPCs")
                if snpcs and snpcs:FindFirstChild("GLOBAL") then return true end
            end
            if attempt < 8 then
                notify("Van attempt "..attempt.."/8, retrying...", GAR.amber, 3, "system")

        else

        end

local function speedCouldMatch(g, spd)
    if spd < 11.5 then return true end
    if thayeConfirmed then return g.name == "Thaye" end
    if spd >= 13.9 and spd <= 14.3 then return g.name == "Thaye" end
    if spd >= 14.8 and spd <= 15.3 then
        if g.name == "Duppy" or g.name == "Wendigo" then return true end
    end
    return spd >= (g.spd - SPEED_TOL) and spd <= (g.spd + SPEED_TOL)
end

local function calcCandidates()
    local speedFiltering = currentSpeedNow >= 11.5
    local speedSet = {}
    if speedFiltering then
        for _, g in ipairs(GHOSTS) do
            if speedCouldMatch(g, currentSpeedNow) then speedSet[g.name] = true end
        end
    end
    local REAL_EV = {emf=true, motion=true, orbs=true, fingerprints=true, book=true}
    local evSet = {}
    for _, g in ipairs(GHOSTS) do
        local ok = true
        for ev in pairs(rejectedEv) do
            if REAL_EV[ev] then
                for _, gev in ipairs(g.ev) do if gev == ev then ok = false; break end end
                if not ok then break end
            end
        end
        if ok then
            for ev in pairs(detectedEv) do
                if REAL_EV[ev] then
                    local has = false
                    for _, gev in ipairs(g.ev) do if gev == ev then has = true; break end end
                    if not has then ok = false; break end
                end
            end
        end
        if ok then evSet[g.name] = true end
    end
    local result = {}
    for _, g in ipairs(GHOSTS) do
        local passes
        if speedFiltering and anyEvidence then passes = speedSet[g.name] and evSet[g.name]
        elseif speedFiltering then passes = speedSet[g.name]
        elseif anyEvidence then passes = evSet[g.name]
        else passes = true end
        if passes and detectedTrait then
            if detectedTrait == "imitator" then
                local is = false
                for _, n in ipairs({"Blair","Mimic","Myling","O Tokata","Shade","Wraith"}) do if g.name == n then is = true; break end end
                passes = is
            else
                passes = (g.trait == detectedTrait)
            end
        end
        if passes and g.trait and rejectedTraits[g.trait] then passes = false end
        if passes and rejectedTraits["imitator"] then
            for _, n in ipairs({"Blair","Mimic","Myling","O Tokata","Shade","Wraith"}) do if g.name == n then passes = false; break end end
        end
        if passes then table.insert(result, g.name) end
    end
    if speedFiltering and currentSpeedNow >= 14.8 and currentSpeedNow <= 15.3 then
        local f = {}; for _, n in ipairs(result) do if n ~= "Preta" then table.insert(f, n) end end; result = f
    end
    if speedFiltering and currentSpeedNow > 15.3 and currentSpeedNow <= 15.8 then
        local f = {}; for _, n in ipairs(result) do if n ~= "Duppy" and n ~= "Wendigo" then table.insert(f, n) end end; result = f
    end
    if thayeConfirmed then
        result = {}
        for _, g in ipairs(GHOSTS) do
            if g.name == "Thaye" then
                local ok = true
                for ev in pairs(rejectedEv) do for _, gev in ipairs(g.ev) do if gev == ev then ok = false; break end end; if not ok then break end end
                if ok then table.insert(result, "Thaye") end
                break
            end
        end
    end
    return result
end

local function isPhantomShadeOnly(c)
    if #c ~= 2 then return false end
    local h = {}; for _, n in ipairs(c) do h[n] = true end
    return h["Phantom"] and h["Shade"]
end

local function submitGhostInJournal(ghostName)
    notify("✦  "..ghostName, GAR.purple, 10, "ghost")
    task.wait(0.5)
    pressKey(Enum.KeyCode.J)
    local ghostSel = nil
    local deadline = tick() + 8
    while tick() < deadline do
        local ok, result = pcall(function()
            return playerGui.Journal.Frame.Frame.Selection.Evidence.Right.GhostSelection
        end)
        if ok and result then ghostSel = result; break end
        task.wait(0.15)
    end
    if not ghostSel then
        pressKey(Enum.KeyCode.J)
        selfDestruct(); return
    end
    if NEEDS_SCROLL_DOWN[ghostName] then
        local selAbs = ghostSel.AbsolutePosition
        local selSz  = ghostSel.AbsoluteSize
        local scrollX = selAbs.X + selSz.X / 2
        local scrollY = selAbs.Y + selSz.Y / 2
        for _ = 1, 3 do
            VirtualInputManager:SendMouseWheelEvent(scrollX, scrollY, false, game)
            task.wait(0.12)
        end
        task.wait(0.25)
    end
    local targetFrame = nil
    for _, child in ipairs(ghostSel:GetChildren()) do
        if child.Name == ghostName then targetFrame = child; break end
    end
    if targetFrame then
        local posDeadline = tick() + 5
        local cx, cy
        while tick() < posDeadline do
            local abs = targetFrame.AbsolutePosition
            local sz  = targetFrame.AbsoluteSize
            if abs.X > 0 and abs.Y > 0 then
                cx = abs.X + sz.X / 2
                cy = abs.Y + sz.Y / 2 + START_OFFSET_Y
                break
            end
            task.wait(0.12)
        end
        if cx then
            task.wait(0.4)
            for _ = 1, 15 do
                local abs2 = targetFrame.AbsolutePosition
                local sz2  = targetFrame.AbsoluteSize
                local guesses = targetFrame:FindFirstChild("Guesses")
                local votes = 0
                if guesses then
                    for _, ch in ipairs(guesses:GetChildren()) do
                        if not ch:IsA("UIGridLayout") then votes += 1 end
                    end
                end
                if votes == 1 then break end
                clickAt(abs2.X + sz2.X/2, abs2.Y + sz2.Y/2 + START_OFFSET_Y)
                task.wait(0.5)
            end
        end
        if NEEDS_SCROLL_DOWN[ghostName] then
            local selAbs = ghostSel.AbsolutePosition
            local selSz  = ghostSel.AbsoluteSize
            local scrollX = selAbs.X + selSz.X / 2
            local scrollY = selAbs.Y + selSz.Y / 2
            task.wait(0.1)
            for _ = 1, 5 do
                VirtualInputManager:SendMouseWheelEvent(scrollX, scrollY, true, game)
                task.wait(0.1)
            end
        end
    end
    pressKey(Enum.KeyCode.J)
    task.wait(0.3)
    selfDestruct()
end

local function updateCandidates()
    if identifiedDone then return end
    local candidates = calcCandidates()
    if isPhantomShadeOnly(candidates) and not bookExtendedTimerLaunched then
        bookExtendedTimerLaunched = true
    end
    if #candidates == 0 then
        local fallback = {}
        for _, g in ipairs(GHOSTS) do
            local ok = true
            for ev in pairs(rejectedEv) do for _, gev in ipairs(g.ev) do if gev == ev then ok = false; break end end; if not ok then break end end
            if ok then for ev in pairs(detectedEv) do local has = false; for _, gev in ipairs(g.ev) do if gev == ev then has = true; break end end; if not has then ok = false; break end end end
            if ok and detectedTrait then
                if detectedTrait == "imitator" then
                    local is = false; for _, n in ipairs({"Blair","Mimic","Myling","O Tokata","Shade","Wraith"}) do if g.name == n then is = true; break end end; ok = is
                else ok = (g.trait == detectedTrait) end
            end
            if ok and g.trait and rejectedTraits[g.trait] then ok = false end
            if ok and rejectedTraits["imitator"] then for _, n in ipairs({"Blair","Mimic","Myling","O Tokata","Shade","Wraith"}) do if g.name == n then ok = false; break end end end
            if ok then table.insert(fallback, g.name) end
        end
        if #fallback == 1 then announceAndSubmit(fallback[1]) end
    elseif #candidates == 1 then
        announceAndSubmit(candidates[1])
    end
end

setTrait = function(traitName)
    if detectedTrait then return end
    if rejectedTraits[traitName] then return end
    detectedTrait = traitName
    notify("Trait: "..traitName:sub(1,1):upper()..traitName:sub(2), GAR.amber, 7, "trait")
    updateCandidates()
    task.spawn(function()
        task.wait(15)
        local REAL_EV2 = {emf=true, motion=true, orbs=true, fingerprints=true, book=true}
        local pool = {}
        for _, g in ipairs(GHOSTS) do
            if g.trait == traitName then
                for _, ev in ipairs(g.ev) do if REAL_EV2[ev] then pool[ev] = true end end
            end
        end
        for _, ev in ipairs({"emf","motion","orbs","fingerprints","book"}) do
            if not pool[ev] and not detectedEv[ev] and not rejectedEv[ev] then rejectEvidence(ev) end
        end
    end)
end

rejectTrait = function(traitName)
    if detectedTrait then return end
    if rejectedTraits[traitName] then return end
    rejectedTraits[traitName] = true
    updateCandidates()
end

confirmEvidence = function(key)
    if detectedEv[key] then return end
    detectedEv[key] = true; anyEvidence = true
    notify(EV_NAMES[key] or key, GAR.green, 5, "evidence")
    updateCandidates()
end

rejectEvidence = function(key)
    if detectedEv[key] then return end
    rejectedEv[key] = true
    updateCandidates()
end

        task.wait(0.2)

        task.wait(0.3)

        VirtualInputManager:SendMouseButtonEvent(vp.X/2, vp.Y/2, 0, true, game, 0)
        task.wait(0.08)

        cam.CameraType = Enum.CameraType.Custom
        if identifiedDone then return end

        if identifiedDone then return end

            cam.CameraType = Enum.CameraType.Scriptable
            cam.CFrame = CFrame.lookAt(cam.CFrame.Position, hrp.Position + Vector3.new(0,-2,0))
            task.wait(0.1)
        end
        pressKey(Enum.KeyCode.Two)
        task.wait(0.3)
        vp = workspace.CurrentCamera.ViewportSize
        VirtualInputManager:SendMouseButtonEvent(vp.X/2, vp.Y/2, 0, true, game, 0)
        task.wait(0.08)
        VirtualInputManager:SendMouseButtonEvent(vp.X/2, vp.Y/2, 0, false, game, 0)

        if identifiedDone then return end

        if identifiedDone then return end

        task.wait(0.2)
        cam = workspace.CurrentCamera
        hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")

            cam.CameraType = Enum.CameraType.Scriptable

                for _, m in pairs(src) do
                    if m:IsA("Part") then
                        if colorClose(m.Color, COLOR_MOTION_YES) then yes = true; break
                        elseif colorClose(m.Color, COLOR_TOOTHPASTE) then no = true end
                    end
                end
                if yes then confirmEvidence("motion")
                elseif no and not detectedEv["motion"] then rejectEvidence("motion") end
            end
            task.wait(0.5)
        end
    end)

    local function watchBook(book)
        local function checkPages()
            if detectedEv["book"] then return end
            local left  = book:FindFirstChild("LeftPage")
            local right = book:FindFirstChild("RightPage")
            if (left and #left:GetChildren()>0) or (right and #right:GetChildren()>0) then confirmEvidence("book") end
        end
        checkPages()
        local function connectPage(page)
            if not page then return end
            page.ChildAdded:Connect(function() confirmEvidence("book") end)
        end
        connectPage(book:FindFirstChild("LeftPage"))
        connectPage(book:FindFirstChild("RightPage"))
        book.ChildAdded:Connect(function(child)
            if child.Name == "LeftPage" or child.Name == "RightPage" then connectPage(child); checkPages() end
        end)
    end
    if equipment then
        for _, item in pairs(equipment:GetChildren()) do if item.Name == "Book" then watchBook(item) end end
        equipment.ChildAdded:Connect(function(item) if item.Name == "Book" then watchBook(item) end end)
    end

    local bookStartTime = tick()
    task.delay(140, function()
        if detectedEv["book"] then return end
        if isPhantomShadeOnly(calcCandidates()) then
            task.delay(math.max(180 - (tick()-bookStartTime), 0), function()
                if not detectedEv["book"] then rejectEvidence("book") end
            end)
        else
            rejectEvidence("book")
        end
    end)

    task.spawn(function()
        local startTime = tick(); local transitions = 0; local lastState = nil; local conn
        conn = RunService.Heartbeat:Connect(function()
            if identifiedDone then conn:Disconnect(); return end
            if detectedTrait or rejectedTraits["flicker"] then conn:Disconnect(); return end
            if tick() - startTime > 20 then conn:Disconnect(); rejectTrait("flicker"); return end
            if not ghostBase then return end
            local t = ghostBase.Transparency
            local currentState = t <= 0.05 and "visible" or "hidden"
            if lastState ~= nil and currentState ~= lastState then transitions += 1 end
            lastState = currentState
            if transitions >= 3 then conn:Disconnect(); setTrait("flicker") end
        end)
    end)

    task.spawn(function()
        local function checkForTrickster(model)
            if detectedTrait then return end
            local main = model:FindFirstChild("Main")
            if not main then return end
            if main:FindFirstChildWhichIsA("BodyForce") then setTrait("trickster"); return end
            main.ChildAdded:Connect(function(child)
                if detectedTrait then return end
                if child:IsA("BodyForce") then setTrait("trickster") end
            end)
        end
        if equipment then
            for _, model in pairs(equipment:GetChildren()) do if model:IsA("Model") then checkForTrickster(model) end end
            equipment.ChildAdded:Connect(function(child) if detectedTrait then return end; if child:IsA("Model") then checkForTrickster(child) end end)
        end
    end)

        if identifiedDone or detectedTrait then return end
        if not rejectedTraits["trickster"] then rejectedTraits["trickster"] = true; updateCandidates() end
        if not rejectedTraits["imitator"]  then rejectedTraits["imitator"]  = true; updateCandidates() end
    end)

    task.spawn(function()
        while not identifiedDone do
            if ghost then
                local speed = ghost:GetAttribute("Speed") or 0
                if speed >= 11.0 then
                    if not firstSpeedReading then firstSpeedReading = { speed=speed, time=tick() } end
                    currentSpeedNow = speed
                    if not thayeConfirmed and speed >= 13.9 and speed <= 14.3 then
                        if firstSpeedReading and firstSpeedReading.speed >= 13.5 and firstSpeedReading.speed <= 14.0 then
                            thayeConfirmed = true
                        end
                    end
                    updateCandidates()
                end
            end
            task.wait(0.3)
        end
    end)

    task.spawn(function()
        local lastPos = nil; local lastT = tick()
        while not identifiedDone do
            task.wait(0.2)
            local cands = calcCandidates()
            local wraithIn = false
            for _, n in ipairs(cands) do if n == "Wraith" then wraithIn = true; break end end
            if not wraithIn then lastPos = nil; continue end
            local gp = ghost and ghost.PrimaryPart and ghost.PrimaryPart.Position
            if gp then
                if lastPos then
                    if (tick()-lastT) < 0.5 and (gp-lastPos).Magnitude > 15 then
                        announceAndSubmit("Wraith"); break
                    end
                end
                lastPos = gp; lastT = tick()
            end
        end
    end)

    task.spawn(function()
        local function mimicIn()
            for _, n in ipairs(calcCandidates()) do if n == "Mimic" then return true end end; return false
        end
        local baseConfirmed = false; local baseReadings = 0
        while not identifiedDone do
            task.wait(0.35)
            if not mimicIn() then baseConfirmed=false; baseReadings=0; continue end
            if not ghost then continue end
            local spd = ghost:GetAttribute("Speed") or 0
            if spd < 11.0 then continue end
            if not baseConfirmed then
                if spd >= (12.4-SPEED_TOL) and spd <= (12.4+SPEED_TOL) then
                    baseReadings += 1
                    if baseReadings >= 4 then baseConfirmed = true end
                else baseReadings = 0 end
            else
                if spd > (12.4+SPEED_TOL) and spd >= 13.0 then announceAndSubmit("Mimic"); break end
            end
        end
    end)

    local IMITATOR_KEYWORDS = {"something is wrong","we should leave","open door for van","something wrong","should leave"}
    local function checkMessageForImitator(text)
        if not text or identifiedDone then return end
        local lower = text:lower()
        for _, kw in ipairs(IMITATOR_KEYWORDS) do
            if lower:find(kw, 1, true) then
                if detectedTrait == "imitator" or rejectedTraits["imitator"] then return end
                detectedTrait = "imitator"
                notify("Trait: Imitator", GAR.amber, 7, "trait")
                updateCandidates()
                return
            end
        end
    end
    local TextChatService = game:GetService("TextChatService")
    if TextChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        TextChatService.MessageReceived:Connect(function(message)
            checkMessageForImitator(message.Text or "")
        end)
    end
end
